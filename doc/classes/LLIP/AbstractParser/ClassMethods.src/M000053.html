      def compile
        
        init_compile
        
        #first check the scope
        if @scope.nil? or not @productions.has_key? @scope.to_sym
          raise "You must give a legal scope"
        end
        
        compile_scope
        
        #compile and eval all the productions
        @productions.values.each { |prod| compile_production(prod) }
        
        class_eval(@code)
        @compiled = true
      end
