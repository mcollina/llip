    def read_next
      unless @source.eof?
        @next_char = @source.readchar.chr       
        while @next_char =~ UTF8_MB_PATTERN
          @next_char << @source.readchar.chr    
        end
        
        if @next_char == $-0
          @current_line += 1
          @current_char = -1    
        else
          @current_char += 1
        end
        
        @next_char
      else
        @next_char = nil
      end
    end
