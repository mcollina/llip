    def raise(*args) 
      if args.first.respond_to? :exception or not @scanner.respond_to? :current or @scanner.current == nil
        super(*args)
      else
        error = ParserError.new(@scanner.current,args.shift)
        backtrace = args.shift
        backtrace ||= caller(1)
        error.set_backtrace(backtrace)
        super error
      end
    end
